```typescript
import Dexie, { Table } from 'dexie';

export interface OfflineReport {
  id?: number;
  localId: string; // UUID untuk offline reports
  title: string;
  description: string;
  location: string;
  category?: string;
  urgency?: string;
  status: 'draft' | 'pending' | 'synced';
  imageUrl?: string;
  latitude?: number;
  longitude?: number;
  createdAt: number;
  syncedAt?: number;
}

export interface OfflineSyncQueue {
  id?: number;
  type: 'create_report' | 'update_status' | 'cancel_report';
  data: any;
  retryCount: number;
  createdAt: number;
}

class OfflineDatabase extends Dexie {
  reports!: Table<OfflineReport>;
  syncQueue!: Table<OfflineSyncQueue>;

  constructor() {
    super('LaporInOfflineDB');
    this.version(1).stores({
      reports: '++id, localId, status, createdAt',
      syncQueue: '++id, type, createdAt',
    });
  }
}

export const db = new OfflineDatabase();
```

#### **3. Create Offline Service**

**File: `lib/offline-service.ts`**

```typescript
import { db, OfflineReport, OfflineSyncQueue } from './offline-db';
import { v4 as uuidv4 } from 'uuid';
import api from './api';

export class OfflineService {
  // Check if online
  static isOnline(): boolean {
    return typeof navigator !== 'undefined' && navigator.onLine;
  }

  // Save report offline
  static async saveReportOffline(report: Omit<OfflineReport, 'id' | 'localId' | 'status' | 'createdAt'>): Promise<string> {
    const localId = uuidv4();
    const offlineReport: OfflineReport = {
      ...report,
      localId,
      status: 'draft',
      createdAt: Date.now(),
    };

    await db.reports.add(offlineReport);
    return localId;
  }

  // Sync offline reports when online
  static async syncOfflineReports(): Promise<void> {
    if (!this.isOnline()) {
      return;
    }

    const offlineReports = await db.reports
      .where('status')
      .anyOf(['draft', 'pending'])
      .toArray();

    for (const report of offlineReports) {
      try {
        // Upload report to server
        const response = await api.post('/api/reports', {
          title: report.title,
          description: report.description,
          location: report.location,
          category: report.category,
          urgency: report.urgency,
          imageUrl: report.imageUrl,
          latitude: report.latitude,
          longitude: report.longitude,
        });

        // Mark as synced
        await db.reports.update(report.id!, {
          status: 'synced',
          syncedAt: Date.now(),
        });

        // Add to sync queue for blockchain (if needed)
        await db.syncQueue.add({
          type: 'create_report',
          data: { reportId: response.data.id, localId: report.localId },
          retryCount: 0,
          createdAt: Date.now(),
        });
      } catch (error) {
        console.error('Sync error:', error);
        // Mark as pending for retry
        await db.reports.update(report.id!, {
          status: 'pending',
        });
      }
    }
  }

  // Get offline reports
  static async getOfflineReports(): Promise<OfflineReport[]> {
    return await db.reports
      .where('status')
      .anyOf(['draft', 'pending'])
      .toArray();
  }

  // Delete synced reports (cleanup)
  static async cleanupSyncedReports(): Promise<void> {
    const oneWeekAgo = Date.now() - 7 * 24 * 60 * 60 * 1000;
    await db.reports
      .where('syncedAt')
      .below(oneWeekAgo)
      .delete();
  }
}
```

**File: `components/OfflineIndicator.tsx`**

```typescript
'use client';

import { useState, useEffect } from 'react';
import { WifiOff, Wifi } from 'lucide-react';

export default function OfflineIndicator() {
  const [isOnline, setIsOnline] = useState(true);
  const [showBanner, setShowBanner] = useState(false);

  useEffect(() => {
    const handleOnline = () => {
      setIsOnline(true);
      setShowBanner(true);
      setTimeout(() => setShowBanner(false), 3000);
    };

    const handleOffline = () => {
      setIsOnline(false);
      setShowBanner(true);
    };

    setIsOnline(navigator.onLine);
    window.addEventListener('online', handleOnline);
    window.addEventListener('offline', handleOffline);

    return () => {
      window.removeEventListener('online', handleOnline);
      window.removeEventListener('offline', handleOffline);
    };
  }, []);

  if (!showBanner) return null;

  return (
    <div
      className={`fixed top-0 left-0 right-0 z-50 p-4 ${
        isOnline ? 'bg-green-500' : 'bg-yellow-500'
      } text-white text-center`}
    >
      <div className="flex items-center justify-center gap-2">
        {isOnline ? (
          <>
            <Wifi className="w-5 h-5" />
            <span>Koneksi internet kembali tersedia</span>
          </>
        ) : (
          <>
            <WifiOff className="w-5 h-5" />
            <span>Mode offline aktif - Laporan akan disinkronkan saat online</span>
          </>
        )}
      </div>
    </div>
  );
}
```

### **Phase 4: Auto-Sync Mechanism (2-3 jam)**

#### **1. Create Sync Hook**

**File: `hooks/useOfflineSync.ts`**

```typescript
'use client';

import { useEffect } from 'react';
import { OfflineService } from '@/lib/offline-service';

export function useOfflineSync() {
  useEffect(() => {
    // Sync on mount if online
    if (OfflineService.isOnline()) {
      OfflineService.syncOfflineReports();
    }

    // Sync when coming online
    const handleOnline = () => {
      OfflineService.syncOfflineReports();
    };

    window.addEventListener('online', handleOnline);

    // Periodic sync (every 30 seconds when online)
    const syncInterval = setInterval(() => {
      if (OfflineService.isOnline()) {
        OfflineService.syncOfflineReports();
      }
    }, 30000);

    return () => {
      window.removeEventListener('online', handleOnline);
      clearInterval(syncInterval);
    };
  }, []);
}
```
